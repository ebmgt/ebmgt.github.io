# This will get executed each time the exercise gets initialized # Per example of Myrna Anne Yeakle(cited below), in her 1975 studies of three interventions, including two different interventions with the Propoganda Game to teach "irrelevance." # ----------------------------- # Table 4.5 # ANCOVA / ANOVA with Course Grade as covariate # ----------------------------- # Testing for basleine differences between groups message("\nTable 4.5. Testing for baseline differences between groups with Course Grade as covariate.\n") # Covariate: Course Grade message("Covariate: Course Grade") F_grade <- 2.38 df1_grade <- 1 df2_grade <- 137 p_grade <- pf(F_grade, df1 = df1_grade, df2 = df2_grade, lower.tail = FALSE) p_grade # Main effect: Group message("Main effect: Experimental Group") F_group_45 <- 3.72 df1_group_45 <- 2 df2_group_45 <- 137 p_group_45 <- pf(F_group_45, df1 = df1_group_45, df2 = df2_group_45, lower.tail = FALSE) p_group_45 message("\nSummary: after controlling for course grade, groups differ significantly at baseline for pretest scores.\n") # ----------------------------- # Table 4.6 # ANCOVA with Year as covariate # ----------------------------- message("\nTable 4.6. Testing for baseline differences between groups with Year in College as a covariate.\n") # Covariate: Year in College message(("Covariate: Year in College")) F_year <- 2.28 df1_year <- 1 df2_year <- 105 p_year <- pf(F_year, df1 = df1_year, df2 = df2_year, lower.tail = FALSE) p_year # Main effect: Group message(("Main effect: Experimental Group")) F_group_46 <- 1.96 df1_group_46 <- 2 df2_group_46 <- 105 p_group_46 <- pf(F_group_46, df1 = df1_group_46, df2 = df2_group_46, lower.tail = FALSE) p_group_46 message("\nSummary: after controlling for year in college, groups did not differ significantly at baseline for pretest scores.\n") test_object("a") test_function("print") success_msg("Great job!")
